🔐 ENDPOINTS DE API
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

POST /api/auth/login
   Request:  { email, password }
   Response: { success, user }
   Errors:   Validación, credenciales incorrectas

POST /api/auth/register
   Request:  { email, password, confirmPassword }
   Response: { success, user }
   Errors:   Validación, email duplicado
